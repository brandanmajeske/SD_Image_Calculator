        private function submitInfo(event:MouseEvent):void
        {

            {    
        
                var gb:Number = Number(_csInput.text);
                var mpix:Number = Number(_mpixInput.text);

            }
            
        }
        
        private function rawJpegCalc(mpix:Number, gb:Number):Number
        {
            
            // Approx JPEG (compressed) file size shot with 10 megapixel cam era is 4.5MB, or .45MB per megapixel.
            // Approx RAW (uncompressed) size shot with 10 megapixel camera is 9MB, or .9MB per megapixel.
            // 1000 Megabytes = 1 Gigabyte
            // calculate how many raw or jpeg per Gigabyte based on number of Megapixels entered by user, and if user is compressing files (jpeg vs. raw)
            // return the number of files based on results of calculation and if statement.
            
            var jpegPerGb:Number = 1000 / (.45 * mpix);
            var jpeg:uint = Math.round(jpegPerGb * gb);
            
            
            var rawPerGb:Number = 1000 / (.90 * mpix);
            var raw:uint = Math.round(rawPerGb * gb);

            
            
            if (_compressInput.text.length <= 2)
            {
                
                trace("Compress No");
                return raw;
                
            } else {    
                trace("Compress Yes");
                return jpeg;
            }    
            
        }
